import { useCallback, useEffect, useRef, useState } from 'react';

type PaddingMode = 'uniform' | 'sides';
type PaddingType = 'transparent' | 'colored';
type ImageFormat = 'png' | 'webp' | 'jpg' | 'svg';
type ThemeMode = 'light' | 'dark' | 'system';

interface PaddingSides {
  top: number;
  right: number;
  bottom: number;
  left: number;
}

export interface SnapSettings {
  padding: number;
  paddingSides: PaddingSides;
  paddingMode: PaddingMode;
  paddingType: PaddingType;
  paddingColor: string;
  captureMargin: number;
  roundedRadius: number;
  squircleRounding: boolean;
  cornerSmoothing: number;
  format: ImageFormat;
  quality: number;
  filenamePrefix: string;
  theme: ThemeMode;
  panelOpacityLow: boolean;
}

export interface Rect {
  x: number;
  y: number;
  width: number;
  height: number;
}

export interface HiddenElement {
  element: HTMLElement;
  label: string;
}

interface ElementSnapPanelProps {
  isActive: boolean;
  isLocked: boolean;
  hiddenElements: HiddenElement[];
  settings: SnapSettings;
  onToggleActive: () => void;
  onToggleLock: () => void;
  onToggleDim: () => void;
  onCapture: () => void;
  onSettingChange: (key: SettingKey, value: SettingValue) => void;
  onRestoreElement: (index: number) => void;
  onRestoreAll: () => void;
}

type FlatSettingKey = keyof SnapSettings;
type NestedSettingKey = `paddingSides.${keyof PaddingSides}`;
type SettingKey = FlatSettingKey | NestedSettingKey;

type SettingValue = number | string | boolean | PaddingSides | ThemeMode | ImageFormat;

const PADDING_SIDES = ['top', 'right', 'bottom', 'left'] as const;
type PaddingSide = (typeof PADDING_SIDES)[number];

const DEFAULTS: Readonly<SnapSettings> = {
  padding: 0,
  paddingSides: { top: 0, right: 0, bottom: 0, left: 0 },
  paddingMode: 'uniform',
  paddingType: 'transparent',
  paddingColor: '#ffffff',
  captureMargin: 0,
  roundedRadius: 0,
  squircleRounding: false,
  cornerSmoothing: 0.6,
  format: 'png',
  quality: 90,
  filenamePrefix: 'element-screenshot',
  theme: 'system',
  panelOpacityLow: false,
};

function ElementSnapPanel({
  isActive,
  isLocked,
  //   currentRect,
  hiddenElements,
  settings,
  onToggleActive,
  onToggleLock,
  onToggleDim,
  onCapture,
  onSettingChange,
  //   onHideElement,
  onRestoreElement,
  onRestoreAll,
}: ElementSnapPanelProps) {
  const supportsAlpha = (fmt: ImageFormat) => fmt === 'png' || fmt === 'webp' || fmt === 'svg';
  const isAlpha = supportsAlpha(settings.format);
  const isUniform = settings.paddingMode === 'uniform';
  const showQuality = settings.format === 'jpg' || settings.format === 'webp';

  return (
    <div className={`fixed w-[280px] z-2147483646 transition-opacity duration-150 ${settings.panelOpacityLow ? 'opacity-10 hover:opacity-100' : 'opacity-100'}`} style={{ pointerEvents: 'auto' }}>
      <div className="bg-white dark:bg-gray-800 border border-gray-200 dark:border-gray-700 rounded-xl shadow-lg p-3 max-h-[72vh] overflow-auto">
        {/* Header */}
        <div className="flex items-center justify-between mb-3">
          <h3 className="font-semibold text-sm text-gray-900 dark:text-gray-100">Screenshot Settings</h3>
          <div className="flex gap-2">
            <button
              onClick={onToggleLock}
              className={`inline-flex items-center justify-center h-7 px-3 rounded-lg border text-xs font-medium transition-colors ${
                isLocked
                  ? 'bg-app-600 text-white border-app-600 hover:bg-app-700'
                  : 'bg-white dark:bg-gray-800 text-gray-700 dark:text-gray-300 border-gray-300 dark:border-gray-600 hover:bg-gray-50 dark:hover:bg-gray-700'
              }`}
            >
              {isLocked ? <>Lock</> : <>Un-Lock</>}
              {isLocked ? 'Locked' : 'Lock'}
            </button>
            <button
              onClick={onToggleDim}
              className="inline-flex items-center justify-center h-7 px-3 rounded-lg border bg-transparent text-gray-700 dark:text-gray-300 border-gray-300 dark:border-gray-600 hover:bg-gray-50 dark:hover:bg-gray-700 text-xs font-medium transition-colors"
            >
              {settings.panelOpacityLow ? <>Eye On</> : <>Eye Off</>}
            </button>
            <button
              onClick={onToggleActive}
              className="inline-flex items-center justify-center h-7 px-3 rounded-lg border bg-transparent text-gray-700 dark:text-gray-300 border-gray-300 dark:border-gray-600 hover:bg-gray-50 dark:hover:bg-gray-700 text-xs font-medium transition-colors"
            >
              X
            </button>
          </div>
        </div>

        {/* Padding Mode */}
        <label className="block text-xs font-medium text-gray-600 dark:text-gray-400 mt-2 mb-1">Padding Mode</label>
        <div className="flex gap-2">
          <button
            onClick={() => onSettingChange('paddingMode', 'uniform')}
            className={`flex-1 h-7 px-3 rounded-lg border text-xs font-medium transition-colors ${
              isUniform
                ? 'bg-app-600 text-white border-app-600'
                : 'bg-white dark:bg-gray-800 text-gray-700 dark:text-gray-300 border-gray-300 dark:border-gray-600 hover:bg-gray-50 dark:hover:bg-gray-700'
            }`}
          >
            Uniform
          </button>
          <button
            onClick={() => onSettingChange('paddingMode', 'sides')}
            className={`flex-1 h-7 px-3 rounded-lg border text-xs font-medium transition-colors ${
              !isUniform
                ? 'bg-app-600 text-white border-app-600'
                : 'bg-white dark:bg-gray-800 text-gray-700 dark:text-gray-300 border-gray-300 dark:border-gray-600 hover:bg-gray-50 dark:hover:bg-gray-700'
            }`}
          >
            Per Side
          </button>
        </div>

        {/* Padding Controls */}
        {isUniform ? (
          <div className="mt-3">
            <label className="block text-xs font-medium text-gray-600 dark:text-gray-400 mb-1">
              Padding: <span className="text-gray-900 dark:text-gray-100">{settings.padding}px</span>
            </label>
            <input
              type="range"
              min="0"
              max="50"
              step="1"
              value={settings.padding}
              onChange={(e) => onSettingChange('padding', Number(e.target.value))}
              className="w-full h-1.5 bg-gray-200 dark:bg-gray-700 rounded-lg appearance-none cursor-pointer accent-app-600"
            />
          </div>
        ) : (
          <div className="grid grid-cols-2 gap-3 mt-3">
            {PADDING_SIDES.map((side) => (
              <div key={side}>
                <label className="block text-xs font-medium text-gray-600 dark:text-gray-400 mb-1 capitalize">
                  {side}: <span className="text-gray-900 dark:text-gray-100">{settings.paddingSides[side]}px</span>
                </label>
                <input
                  type="range"
                  min="0"
                  max="50"
                  step="1"
                  value={settings.paddingSides[side]}
                  onChange={(e) => onSettingChange(`paddingSides.${side}`, Number(e.target.value))}
                  className="w-full h-1.5 bg-gray-200 dark:bg-gray-700 rounded-lg appearance-none cursor-pointer accent-app-600"
                />
              </div>
            ))}
          </div>
        )}

        {/* Capture Margin */}
        <div className="mt-3">
          <label className="block text-xs font-medium text-gray-600 dark:text-gray-400 mb-1">
            Capture Margin: <span className="text-gray-900 dark:text-gray-100">{settings.captureMargin}px</span>
          </label>
          <input
            type="range"
            min="0"
            max="200"
            step="2"
            value={settings.captureMargin}
            onChange={(e) => onSettingChange('captureMargin', Number(e.target.value))}
            className="w-full h-1.5 bg-gray-200 dark:bg-gray-700 rounded-lg appearance-none cursor-pointer accent-app-600"
          />
        </div>

        {/* Transparency/Color Controls */}
        {isAlpha ? (
          <div className="mt-3">
            <div className="flex gap-2">
              <button
                onClick={() => onSettingChange('paddingType', 'transparent')}
                className={`flex-1 h-7 px-3 rounded-lg border text-xs font-medium transition-colors ${
                  settings.paddingType === 'transparent'
                    ? 'bg-app-600 text-white border-app-600'
                    : 'bg-white dark:bg-gray-800 text-gray-700 dark:text-gray-300 border-gray-300 dark:border-gray-600 hover:bg-gray-50 dark:hover:bg-gray-700'
                }`}
              >
                Transparent
              </button>
              <button
                onClick={() => onSettingChange('paddingType', 'colored')}
                className={`flex-1 h-7 px-3 rounded-lg border text-xs font-medium transition-colors ${
                  settings.paddingType === 'colored'
                    ? 'bg-app-600 text-white border-app-600'
                    : 'bg-white dark:bg-gray-800 text-gray-700 dark:text-gray-300 border-gray-300 dark:border-gray-600 hover:bg-gray-50 dark:hover:bg-gray-700'
                }`}
              >
                Colored
              </button>
            </div>
            {settings.paddingType === 'colored' && (
              <>
                <label className="block text-xs font-medium text-gray-600 dark:text-gray-400 mt-2 mb-1">Padding Color</label>
                <input
                  type="color"
                  value={settings.paddingColor}
                  onChange={(e) => onSettingChange('paddingColor', e.target.value)}
                  className="w-full h-7 rounded-lg border border-gray-300 dark:border-gray-600 cursor-pointer"
                />
              </>
            )}
          </div>
        ) : (
          <div className="mt-3">
            <p className="text-xs text-gray-500 dark:text-gray-400 mb-2">Format has no transparency; using Background Color for export.</p>
            <label className="block text-xs font-medium text-gray-600 dark:text-gray-400 mb-1">Background Color</label>
            <input
              type="color"
              value={settings.paddingColor}
              onChange={(e) => onSettingChange('paddingColor', e.target.value)}
              className="w-full h-7 rounded-lg border border-gray-300 dark:border-gray-600 cursor-pointer"
            />
          </div>
        )}

        {/* Rounded Corners */}
        <div className="mt-3">
          <label className="block text-xs font-medium text-gray-600 dark:text-gray-400 mb-1">
            Rounded Corners: <span className="text-gray-900 dark:text-gray-100">{settings.roundedRadius}px</span>
          </label>
          <input
            type="range"
            min="0"
            max="48"
            step="1"
            value={settings.roundedRadius}
            onChange={(e) => onSettingChange('roundedRadius', Number(e.target.value))}
            className="w-full h-1.5 bg-gray-200 dark:bg-gray-700 rounded-lg appearance-none cursor-pointer accent-app-600"
          />
          <label className="flex items-center gap-2 mt-2 cursor-pointer select-none">
            <input
              type="checkbox"
              checked={settings.squircleRounding}
              onChange={(e) => onSettingChange('squircleRounding', e.target.checked)}
              className="w-4 h-4 rounded border-gray-300 dark:border-gray-600 text-app-600 focus:ring-2 focus:ring-app-500 dark:bg-gray-700"
            />
            <span className="text-xs text-gray-600 dark:text-gray-400">Smooth corners</span>
          </label>
          {settings.squircleRounding && (
            <div className="mt-2">
              <label className="block text-xs font-medium text-gray-600 dark:text-gray-400 mb-1">
                Smoothing: <span className="text-gray-900 dark:text-gray-100">{Math.round(settings.cornerSmoothing * 100)}%</span>
              </label>
              <input
                type="range"
                min="0"
                max="100"
                step="5"
                value={Math.round(settings.cornerSmoothing * 100)}
                onChange={(e) => onSettingChange('cornerSmoothing', Number(e.target.value) / 100)}
                className="w-full h-1.5 bg-gray-200 dark:bg-gray-700 rounded-lg appearance-none cursor-pointer accent-app-600"
              />
            </div>
          )}
        </div>

        {/* Format */}
        <div className="mt-3">
          <label className="block text-xs font-medium text-gray-600 dark:text-gray-400 mb-1">Format</label>
          <select
            value={settings.format}
            onChange={(e) => onSettingChange('format', e.target.value)}
            className="w-full h-7 px-2 rounded-lg border border-gray-300 dark:border-gray-600 bg-white dark:bg-gray-800 text-gray-900 dark:text-gray-100 text-xs focus:ring-2 focus:ring-app-500 focus:border-app-500"
          >
            <option value="png">PNG</option>
            <option value="webp">WEBP</option>
            <option value="jpg">JPG</option>
            <option value="svg">SVG</option>
          </select>
        </div>

        {showQuality && (
          <div className="mt-3">
            <label className="block text-xs font-medium text-gray-600 dark:text-gray-400 mb-1">
              Quality: <span className="text-gray-900 dark:text-gray-100">{settings.quality}%</span>
            </label>
            <input
              type="range"
              min="10"
              max="100"
              step="5"
              value={settings.quality}
              onChange={(e) => onSettingChange('quality', Number(e.target.value))}
              className="w-full h-1.5 bg-gray-200 dark:bg-gray-700 rounded-lg appearance-none cursor-pointer accent-app-600"
            />
          </div>
        )}

        {/* Filename */}
        <div className="mt-3">
          <label className="block text-xs font-medium text-gray-600 dark:text-gray-400 mb-1">Filename Prefix</label>
          <input
            type="text"
            value={settings.filenamePrefix}
            onChange={(e) => onSettingChange('filenamePrefix', e.target.value)}
            className="w-full h-7 px-2 rounded-lg border border-gray-300 dark:border-gray-600 bg-white dark:bg-gray-800 text-gray-900 dark:text-gray-100 text-xs focus:ring-2 focus:ring-app-500 focus:border-app-500"
          />
        </div>

        {/* Hidden Elements */}
        <div className="mt-4">
          <label className="block text-xs font-medium text-gray-600 dark:text-gray-400 mb-1">Clean up</label>
          <p className="text-xs text-gray-500 dark:text-gray-400 mb-2">
            Hidden elements: {hiddenElements.length} - Press{' '}
            <kbd className="px-1.5 py-0.5 bg-gray-100 dark:bg-gray-700 border border-gray-300 dark:border-gray-600 rounded text-xs font-semibold">H</kbd> to hide current,{' '}
            <kbd className="px-1.5 py-0.5 bg-gray-100 dark:bg-gray-700 border border-gray-300 dark:border-gray-600 rounded text-xs font-semibold">R</kbd> to restore last
          </p>
          <div className="max-h-40 overflow-auto space-y-2">
            {hiddenElements.length === 0 ? (
              <p className="text-xs text-gray-500 dark:text-gray-400">No hidden elements yet.</p>
            ) : (
              hiddenElements.map((item, idx) => (
                <div key={idx} className="flex items-center justify-between gap-2">
                  <span className="text-xs text-gray-600 dark:text-gray-400 truncate flex-1" title={item.label}>
                    {item.label}
                  </span>
                  <button
                    onClick={() => onRestoreElement(idx)}
                    className="h-6 px-2 rounded-lg border border-gray-300 dark:border-gray-600 bg-white dark:bg-gray-800 text-gray-700 dark:text-gray-300 hover:bg-gray-50 dark:hover:bg-gray-700 text-xs font-medium transition-colors"
                  >
                    <>Undo</>
                  </button>
                </div>
              ))
            )}
          </div>
          {hiddenElements.length > 0 && (
            <button
              onClick={onRestoreAll}
              className="w-full mt-2 h-7 px-3 rounded-lg border border-gray-300 dark:border-gray-600 bg-white dark:bg-gray-800 text-gray-700 dark:text-gray-300 hover:bg-gray-50 dark:hover:bg-gray-700 text-xs font-medium transition-colors flex items-center justify-center gap-1"
            >
              Restore All
            </button>
          )}
        </div>

        {/* Shortcuts */}
        <div className="mt-4 space-y-1.5 text-xs text-gray-600 dark:text-gray-400">
          <div className="flex items-center gap-2">
            <kbd className="px-1.5 py-0.5 bg-gray-100 dark:bg-gray-700 border border-gray-300 dark:border-gray-600 rounded font-semibold">Ctrl/Cmd + Click</kbd>
            <span>Capture</span>
          </div>
          <div className="flex items-center gap-2">
            <kbd className="px-1.5 py-0.5 bg-gray-100 dark:bg-gray-700 border border-gray-300 dark:border-gray-600 rounded font-semibold">L</kbd>
            <span>Lock/Unlock</span>
          </div>
          <div className="flex items-center gap-2">
            <kbd className="px-1.5 py-0.5 bg-gray-100 dark:bg-gray-700 border border-gray-300 dark:border-gray-600 rounded font-semibold">Esc</kbd>
            <span>Unlock</span>
          </div>
        </div>

        {/* Action Buttons */}
        <div className="flex gap-2 mt-4">
          <button
            onClick={onToggleActive}
            className="flex-1 h-8 px-3 rounded-lg border border-gray-300 dark:border-gray-600 bg-transparent text-gray-700 dark:text-gray-300 hover:bg-gray-50 dark:hover:bg-gray-700 text-xs font-medium transition-colors"
          >
            Close
          </button>
          <button onClick={onCapture} className="flex-1 h-8 px-3 rounded-lg bg-app-600 text-white hover:bg-app-700 text-xs font-medium transition-colors flex items-center justify-center gap-1">
            Capture
          </button>
        </div>
      </div>
    </div>
  );
}

export default function ElementSnapExtension() {
  const [isActive, setIsActive] = useState<boolean>(false);
  const [isLocked, setIsLocked] = useState<boolean>(false);
  const [currentTarget, setCurrentTarget] = useState<HTMLElement | null>(null);
  const [currentRect, setCurrentRect] = useState<Rect | null>(null);
  const [hiddenElements, setHiddenElements] = useState<HiddenElement[]>([]);
  const [settings, setSettings] = useState<SnapSettings>(DEFAULTS);
  const [panelPosition, setPanelPosition] = useState({ left: 16, top: 16 });

  const overlayRef = useRef<HTMLDivElement | null>(null);
  const panelRef = useRef<HTMLDivElement | null>(null);

  const handleSettingChange = useCallback((key: SettingKey, value: SettingValue) => {
    setSettings((prev) => {
      if (key.startsWith('paddingSides.')) {
        const side = key.split('.')[1] as keyof PaddingSides;
        return {
          ...prev,
          paddingSides: {
            ...prev.paddingSides,
            [side]: value as number,
          },
        };
      }

      if (key === 'padding') {
        const v = value as number;
        return {
          ...prev,
          padding: v,
          paddingSides: { top: v, right: v, bottom: v, left: v },
        };
      }

      return { ...prev, [key]: value } as SnapSettings;
    });
  }, []);

  const handleCapture = useCallback(() => {
    console.log('Capture initiated', { currentRect, settings });
    // Implementation would integrate with chrome.tabs.captureVisibleTab
    // This is a placeholder for the WXT extension implementation
  }, [currentRect, settings]);

  const handleHideElement = useCallback(() => {
    if (!currentTarget) return;

    const label = currentTarget.tagName.toLowerCase() + (currentTarget.id ? `#${currentTarget.id}` : '') + (currentTarget.className ? `.${currentTarget.className.split(' ')[0]}` : '');

    setHiddenElements((prev) => [...prev, { element: currentTarget, label }]);
    currentTarget.style.display = 'none';
    setCurrentTarget(null);
    setCurrentRect(null);
  }, [currentTarget]);

  const handleRestoreElement = useCallback(
    (index: number) => {
      const item = hiddenElements[index];
      if (item?.element) {
        item.element.style.display = '';
      }
      setHiddenElements((prev) => prev.filter((_, i) => i !== index));
    },
    [hiddenElements]
  );

  const handleRestoreAll = useCallback(() => {
    hiddenElements.forEach((item) => {
      if (item?.element) {
        item.element.style.display = '';
      }
    });
    setHiddenElements([]);
  }, [hiddenElements]);

  // Keyboard shortcuts
  useEffect(() => {
    if (!isActive) return;

    const handleKeyDown = (e: KeyboardEvent) => {
      if (e.key === 'l' || e.key === 'L') {
        setIsLocked((prev) => !prev);
      } else if (e.key === 'Escape' && isLocked) {
        setIsLocked(false);
      } else if (e.key === 'h' || e.key === 'H') {
        handleHideElement();
      } else if (e.key === 'r' || e.key === 'R') {
        if (e.shiftKey) {
          handleRestoreAll();
        } else {
          handleRestoreElement(hiddenElements.length - 1);
        }
      }
    };

    window.addEventListener('keydown', handleKeyDown);
    return () => window.removeEventListener('keydown', handleKeyDown);
  }, [isActive, isLocked, handleHideElement, handleRestoreElement, handleRestoreAll, hiddenElements.length]);

  // Mouse tracking
  useEffect(() => {
    if (!isActive || isLocked) return;

    const handleMouseMove = (e: MouseEvent) => {
      const el = document.elementFromPoint(e.clientX, e.clientY) as HTMLElement;
      if (el && el !== document.body && el !== document.documentElement) {
        setCurrentTarget(el);
        const rect = el.getBoundingClientRect();
        setCurrentRect({
          x: rect.left,
          y: rect.top,
          width: rect.width,
          height: rect.height,
        });
      }
    };

    window.addEventListener('mousemove', handleMouseMove);
    return () => window.removeEventListener('mousemove', handleMouseMove);
  }, [isActive, isLocked]);

  if (!isActive) {
    return (
      <div className="fixed bottom-4 right-4 z-2147483647">
        <button onClick={() => setIsActive(true)} className="px-4 py-2 bg-app-600 text-white rounded-lg shadow-lg hover:bg-app-700 transition-colors flex items-center gap-2">
          Enable Element Snap
        </button>
      </div>
    );
  }

  return (
    <>
      {/* Overlay */}
      <div ref={overlayRef} className="fixed inset-0 pointer-events-none z-2147483640">
        {currentRect && (
          <>
            {/* Element outline */}
            <div
              className="fixed border-2 border-app-600 outline-2 outline-app-600/25 outline-offset-2 shadow-[0_0_0_4px_rgba(99,102,241,0.15)] transition-all duration-75"
              style={{
                left: `${currentRect.x}px`,
                top: `${currentRect.y}px`,
                width: `${currentRect.width}px`,
                height: `${currentRect.height}px`,
                zIndex: 2147483644,
              }}
            />

            {/* Padding preview - simplified for demo */}
            {(settings.padding > 0 || Object.values(settings.paddingSides).some((v) => v > 0)) && (
              <div
                className="fixed border-2 border-dashed border-app-400 pointer-events-none"
                style={{
                  left: `${currentRect.x - (settings.paddingMode === 'uniform' ? settings.padding : settings.paddingSides.left)}px`,
                  top: `${currentRect.y - (settings.paddingMode === 'uniform' ? settings.padding : settings.paddingSides.top)}px`,
                  width: `${currentRect.width + (settings.paddingMode === 'uniform' ? settings.padding * 2 : settings.paddingSides.left + settings.paddingSides.right)}px`,
                  height: `${currentRect.height + (settings.paddingMode === 'uniform' ? settings.padding * 2 : settings.paddingSides.top + settings.paddingSides.bottom)}px`,
                  borderRadius: settings.roundedRadius > 0 ? `${settings.roundedRadius}px` : '0',
                  zIndex: 2147483642,
                }}
              />
            )}
          </>
        )}
      </div>

      {/* Panel */}
      <div
        ref={panelRef}
        style={{
          left: `${panelPosition.left}px`,
          top: `${panelPosition.top}px`,
        }}
      >
        <ElementSnapPanel
          isActive={isActive}
          isLocked={isLocked}
          //   currentRect={currentRect}
          hiddenElements={hiddenElements}
          settings={settings}
          onToggleActive={() => setIsActive(false)}
          onToggleLock={() => setIsLocked((prev) => !prev)}
          onToggleDim={() => handleSettingChange('panelOpacityLow', !settings.panelOpacityLow)}
          onCapture={handleCapture}
          onSettingChange={handleSettingChange}
          //   onHideElement={handleHideElement}
          onRestoreElement={handleRestoreElement}
          onRestoreAll={handleRestoreAll}
        />
      </div>
    </>
  );
}
